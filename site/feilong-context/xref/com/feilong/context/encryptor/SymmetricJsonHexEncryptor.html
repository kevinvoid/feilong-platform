16  package com.feilong.context.encryptor;
17  
18  import static com.feilong.core.CharsetType.UTF8;
19  
20  import org.slf4j.Logger;
21  import org.slf4j.LoggerFactory;
22  
23  import com.feilong.core.CharsetType;
24  import com.feilong.json.jsonlib.JsonUtil;
25  import com.feilong.security.symmetric.SymmetricEncryption;
26  import com.feilong.security.symmetric.SymmetricType;
27  
28  /**
29   * 默认的加密解密器.
30   *
31   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
32   * @param <T>
33   *            the generic type
34   * @since 1.12.0
35   */
36  public class SymmetricJsonHexEncryptor<T extends Encryptorable> implements Encryptor<T>{
37  
38      /** The Constant log. */
39      private static final Logger LOGGER        = LoggerFactory.getLogger(SymmetricJsonHexEncryptor.class);
40  
41      //---------------------------------------------------------------
42      /** 对称加密的类型. */
43      private SymmetricType       symmetricType = SymmetricType.DESede;
44  
45      /** key. */
46      private String              key           = SymmetricJsonHexEncryptor.class.getName();
47  
48      //---------------------------------------------------------------
49  
50      /** 编码格式,默认 {@link CharsetType#UTF8}. */
51      private String              charsetName   = UTF8;
52  
53      //---------------------------------------------------------------
54  
55      /*
56       * (non-Javadoc)
57       * 
58       * @see com.feilong.psi.controller.redirectresult.PayResultParamDataSafe#encryptPayResult(java.io.Serializable)
59       */
60      @Override
61      public String encrypt(T bean){
62          String json = JsonUtil.format(bean, 0, 0);
63  
64          SymmetricEncryption symmetricEncryption = build();
65          return symmetricEncryption.encryptHex(json, charsetName);
66      }
67  
68      //---------------------------------------------------------------
69  
70      /*
71       * (non-Javadoc)
72       * 
73       * @see com.feilong.psi.controller.redirectresult.PayResultParamDataSafe#decryptPayResult(java.lang.String)
74       */
75      @Override
76      public T decrypt(String encryptPayResultType,Class<T> klass){
77          SymmetricEncryption symmetricEncryption = build();
78          String json = symmetricEncryption.decryptHex(encryptPayResultType, charsetName);
79  
80          //---------------------------------------------------------------
81          if (LOGGER.isDebugEnabled()){
82              LOGGER.debug("encryptPayResultType:[{}],json:[{}]", encryptPayResultType, json);
83          }
84          return JsonUtil.toBean(json, klass);
85      }
86  
87      //---------------------------------------------------------------
88  
89      /**
90       * Builds the.
91       *
92       * @return the symmetric encryption
93       */
94      private SymmetricEncryption build(){
95          return new SymmetricEncryption(symmetricType, key);
96      }
97  
98      //---------------------------------------------------------------
99  
100     /**
101      * 设置 对称加密的类型.
102      *
103      * @param symmetricType
104      *            the symmetricType to set
105      */
106     public void setSymmetricType(SymmetricType symmetricType){
107         this.symmetricType = symmetricType;
108     }
109 
110     /**
111      * 设置 key.
112      *
113      * @param key
114      *            the key to set
115      */
116     public void setKey(String key){
117         this.key = key;
118     }
119 
120     /**
121      * 设置 编码格式,默认 {@link CharsetType#UTF8}.
122      *
123      * @param charsetName
124      *            the charsetName to set
125      */
126     public void setCharsetName(String charsetName){
127         this.charsetName = charsetName;
128     }
129 }
