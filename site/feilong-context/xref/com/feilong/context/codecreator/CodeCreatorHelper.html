18  import static org.apache.commons.lang3.StringUtils.EMPTY;
19  import static org.apache.commons.lang3.StringUtils.SPACE;
20  
21  import org.apache.commons.lang3.StringUtils;
22  
23  import com.feilong.core.lang.StringUtil;
24  
25  /**
26   * The Class CodeCreatorHelper.
27   *
28   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
29   * @since 1.10.4
30   */
31  public class CodeCreatorHelper{
32  
33      /**
34       * Debug.
35       *
36       * @param isDebug
37       *            the is debug
38       * @return the string
39       */
40      public static String debugSeparator(boolean isDebug){
41          return isDebug ? SPACE : EMPTY;
42      }
43  
44      /**
45       * 显示长度.
46       *
47       * @param sb
48       *            the sb
49       * @param isDebug
50       *            the is debug
51       * @return the string
52       */
53      public static String debugLength(StringBuilder sb,boolean isDebug){
54          return isDebug ? " (" + (sb.length() - StringUtils.countMatches(sb.toString(), " ")) + ")" : "";
55      }
56  
57      //---------------------------------------------------------------
58  
59      /**
60       * 格式化值.
61       * 
62       * @param value
63       *            the value
64       * @param lastLength
65       *            the last length
66       * @return
67       *         <ul>
68       *         <li>如果 value 的长度> lastLength,那么截取后lastLength 位数</li>
69       *         <li>如果 value 的长度= lastLength,那么直接返回value字符串格式</li>
70       *         <li>如果 value 的长度< lastLength,那么数字前面补0返回</li>
71       *         </ul>
72       */
73      public static String formatLastValue(Long value,int lastLength){
74          String valueString = value.toString();
75          int valueLength = valueString.length();
76  
77          if (valueLength > lastLength){
78              return StringUtil.substringLast(valueString, lastLength);
79          }
80          if (valueLength == lastLength){
81              return valueString;
82          }
83          //valueLength < lastLength
84          return StringUtil.format("%0" + lastLength + "d", value);
85      }
