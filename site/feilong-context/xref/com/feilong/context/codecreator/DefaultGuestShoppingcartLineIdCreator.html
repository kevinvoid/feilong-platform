16  package com.feilong.context.codecreator;
17  
18  import static com.feilong.core.DatePattern.COMMON_DATE;
19  import static com.feilong.core.bean.ConvertUtil.toLong;
20  import static com.feilong.core.date.DateExtensionUtil.getIntervalTime;
21  import static com.feilong.core.date.DateUtil.now;
22  import static com.feilong.core.date.DateUtil.toDate;
23  
24  import java.util.concurrent.atomic.AtomicLong;
25  
26  import com.feilong.core.util.RandomUtil;
27  
28  /**
29   * 默认的游客购物车行id 创造器.
30   *
31   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
32   * @since 1.10.6
33   */
34  public class DefaultGuestShoppingcartLineIdCreator implements GuestShoppingcartLineIdCreator{
35  
36      /**
37       * AtomicLong.
38       * <ul>
39       * <li>使用 2017-01-01 主要为了生成的长度短一些</li>
40       * <li>游客购物车主要是当前用户cookie的增删改,即使和其他用户重复了也没有关系</li>
41       * <li>初始值理论上,每次重启都是不一样的, 且不会重复,只要服务器时间设置是ok的</li>
42       * </ul>
43       */
44      private static AtomicLong ATOMIC_LONG = new AtomicLong(getIntervalTime(now(), toDate("2017-01-01", COMMON_DATE)));
45  
46      //---------------------------------------------------------------
47  
48      /**
49       * 创建 id.
50       * 
51       * <p>
52       * 游客的购物车行id 仅仅用于增删修改等操作,不会存储到DB
53       * </p>
54       *
55       * @return the long
56       * @see <a href="/NB-1074">游客购物车行id规则bug</a>
57       */
58      @Override
59      public long create(){ //shoppingCartLines.size() 不能解决删除的问题
60  
61          StringBuilder sb = new StringBuilder();
62          sb.append(ATOMIC_LONG.getAndIncrement());
63          sb.append(RandomUtil.createRandomWithLength(4));//减少因为服务器时间等场景带来的幺蛾子事情
64  
65          return toLong(sb);
66      }
67  }
