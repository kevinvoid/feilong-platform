42      public void testSortByPropertyNamesValueMap(){
43          Map<String, Integer> map18_145 = toMap("age", 18, "height", 145);
44          Map<String, Integer> map14_120 = toMap("age", 14, "height", 120);
45          Map<String, Integer> map5_45 = toMap("age", 5, "height", 45);
46          Map<String, Integer> map20_170 = toMap("age", 20, "height", 170);
47  
48          List<Map<String, Integer>> list = toList(map18_145, map14_120, map5_45, map20_170);
49          sortListByPropertyNamesValue(list, "age");
50  
51          assertThat(list, contains(map5_45, map14_120, map18_145, map20_170));
52  
53      }
