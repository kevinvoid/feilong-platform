43      /**
44       * Test sort list comparators.
45       */
46      @SuppressWarnings("static-method")
47      @Test
48      public void testSortListComparators(){
49          User zhangfei = new User("张飞", 23);
50          User guanyu = new User("关羽", 30);
51          User liubei = new User("刘备", 25);
52  
53          String[] names = { "刘备", "关羽" };
54          List<User> list = CollectionsUtil.select(toList(zhangfei, guanyu, liubei), "name", names);
55          sortList(list, new PropertyComparator<User>("name", new FixedOrderComparator<>(names)));
56  
57          assertThat(list, contains(liubei, guanyu));
58      }
59  
60      /**
61       * Test sort list comparators 1.
62       */
63      @SuppressWarnings("static-method")
64      @Test
65      public void testSortListComparators1(){
66          User guanyu = new User("关羽", 30);
67  
68          User liubei60 = new User("刘备", 60);
69          User liubei25 = new User("刘备", 25);
70          User liubei30 = new User("刘备", 30);
71          User liubei10 = new User("刘备", 10);
72  
73          String[] names = { "刘备", "关羽" };
74          List<User> list = CollectionsUtil.select(toList(liubei60, liubei30, liubei10, guanyu, liubei25), "name", names);
75          sortList(
76                          list, //
77                          new PropertyComparator<User>("name", new FixedOrderComparator<>(names)),
78                          new PropertyComparator<User>("age"));
79  
80          assertThat(list, contains(liubei10, liubei25, liubei30, liubei60, guanyu));
81      }
