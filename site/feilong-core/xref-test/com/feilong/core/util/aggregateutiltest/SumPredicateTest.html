49      /**
50       * Test sum3.
51       */
52      @Test
53      public void testSum3(){
54          User liubei = new User(10L);
55          liubei.setName("刘备");
56          liubei.setAge(50);
57  
58          User guanyu = new User(20L);
59          liubei.setName("关羽");
60          guanyu.setAge(50);
61  
62          User zhangfei = new User(100L);
63          zhangfei.setName("张飞");
64          zhangfei.setAge(null);
65  
66          User zhaoyun = new User((Long) null);
67          zhaoyun.setName("赵云");
68          zhaoyun.setAge(100);
69  
70          List<User> list = toList(liubei, guanyu, zhangfei, zhaoyun);
71  
72          Predicate<User> notPredicate = PredicateUtils.notPredicate(BeanPredicateUtil.equalPredicate("name", "张飞"));
73          Map<String, BigDecimal> map = AggregateUtil.sum(list, toArray("id", "age"), notPredicate);
74  
75          assertThat(map, allOf(hasEntry("id", toBigDecimal(30)), hasEntry("age", toBigDecimal(200))));
76      }
77
