39  /**
40   * The Class BeanUtilCopyPropertiesTest.
41   *
42   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
43   */
44  public class CopyPropertiesTest{
45  
46      /**
47       * Test copy properties null value.
48       */
49      @Test
50      @SuppressWarnings("static-method")
51      public void testCopyPropertiesNullValue(){
52          User oldUser = new User();
53          oldUser.setId(null);
54  
55          User newUser = new User();
56          BeanUtil.copyProperties(newUser, oldUser);
57  
58          assertEquals(null, newUser.getId());
59      }
60  
61      /**
62       * Test copy properties all.
63       */
64      @Test
65      @SuppressWarnings("static-method")
66      public void testCopyPropertiesAll(){
67          BigDecimal money = new BigDecimal(500000);
68          String[] nickNames = toArray("feilong", "飞天奔月", "venusdrogon");
69          Date date = now();
70  
71          User user = new User();
72          user.setId(5L);
73          user.setMoney(money);
74          user.setDate(date);
75          user.setNickNames(nickNames);
76  
77          ConvertUtil.registerSimpleDateLocaleConverter(TO_STRING_STYLE);
78  
79          User user2 = new User();
80          BeanUtil.copyProperties(user2, user);
81  
82          //---------------------------------------------------------------
83          assertThat(
84                          user2,
85                          allOf(//
86                                          hasProperty("money", is(money)),
87  
88                                          hasProperty("nickNames", is(nickNames)),
89                                          //hasProperty("nickNames", is(toArray("feilong"))),
90  
91                                          // not(hasProperty("date", equalTo(date)))//
92                                          hasProperty("date", equalTo(date))//
93                          ));
94      }
95  
96      /**
97       * Test copy properties with array.
98       */
99      @Test
100     @SuppressWarnings("static-method")
101     public void testCopyPropertiesWithArray(){
102         String[] nickNames = toArray("feilong", "飞天奔月", "venusdrogon");
103 
104         User user = new User();
105         user.setNickNames(nickNames);
106 
107         User user2 = new User();
108         BeanUtil.copyProperties(user2, user, "nickNames");
109 
110         assertThat(
111                         user2,
112                         allOf(//
113                                         not(hasProperty("nickNames", is(nickNames))),
114                                         hasProperty("nickNames", is(toArray("feilong")))));
115     }
116 
117     /**
118      * Test copy property to bean not exist properties.
119      */
120     @Test
121     @SuppressWarnings("static-method")
122     public void testCopyPropertyToBeanNotExistProperties(){
123         User user = new User();
124         user.setId(5L);
125 
126         Person person = new Person();
127         BeanUtil.copyProperties(person, user, "age", "name");
128 
129         assertThat(person, allOf(hasProperty("name", is("feilong")), hasProperty("dateAttr", is((Object) null))));
130     }
