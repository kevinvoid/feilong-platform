16  package com.feilong.core.net.paramutiltest;
17  
18  import static com.feilong.core.CharsetType.UTF8;
19  import static com.feilong.core.net.ParamUtil.addParameter;
20  import static com.feilong.core.net.URIUtil.encode;
21  import static org.apache.commons.lang3.StringUtils.EMPTY;
22  import static org.junit.Assert.assertEquals;
23  
24  import org.junit.Test;
25  
26  /**
27   * The Class ParamUtilAddParameterTest.
28   *
29   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
30   */
31  public class AddParameterTest{
32  
33      /** The Constant PATH. */
34      private static final String PATH = "http://www.feilong.com:8888/esprit-frontend/search.htm";
35  
36      /**
37       * Test add parameter no param.
38       */
39      @Test
40      public void testAddParameterNoParam(){
41          String uriString = PATH;
42          assertEquals(PATH + "?label=2-5-8-12", addParameter(uriString, "label", "2-5-8-12", UTF8));
43      }
44  
45      /**
46       * Test add parameter.
47       */
48      @Test
49      public void testAddParameter(){
50          String uriString = PATH + "?keyword=%E6%81%A4&page=";
51          assertEquals(PATH + "?keyword=%E6%81%A4&page=&label=2-5-8-12", addParameter(uriString, "label", "2-5-8-12", UTF8));
52      }
53  
54      /**
55       * Test add parameter chinese value.
56       */
57      @Test
58      public void testAddParameterChineseValue(){
59          String uriString = PATH + "?keyword=%E6%81%A4&page=";
60          String value = "中国";
61          assertEquals(PATH + "?keyword=%E6%81%A4&page=&label=" + encode(value, UTF8), addParameter(uriString, "label", value, UTF8));
62      }
63  
64      /**
65       * Test add parameter replace value.
66       */
67      @Test
68      public void testAddParameterReplaceValue(){
69          String uriString = PATH + "?keyword=%E6%81%A4&label=hahaha&page=";
70          String value = "中国";
71          assertEquals(PATH + "?keyword=%E6%81%A4&label=" + encode(value, UTF8) + "&page=", addParameter(uriString, "label", value, UTF8));
72      }
73  
74      /**
75       * Test add parameter replace value two params.
76       */
77      @Test
78      public void testAddParameterReplaceValueTwoParams(){
79          String uriString = PATH + "?label=lalala&keyword=%E6%81%A4&label=hahaha&page=";
80  
81          String value = "中国";
82          String expected = PATH + "?label=" + encode(value, UTF8) + "&keyword=%E6%81%A4&page=";
83          assertEquals(expected, addParameter(uriString, "label", value, UTF8));
84      }
85  
86      /**
87       * Test add parameter null charset type.
88       */
89      @Test
90      public void testAddParameterNullCharsetType(){
91          String uriString = PATH + "?keyword=%E6%81%A4&page=";
92          assertEquals(PATH + "?keyword=%E6%81%A4&page=&label=中国", addParameter(uriString, "label", "中国", null));
93      }
94  
95      /**
96       * Test add parameter null param name.
97       */
98      @Test
99      public void testAddParameterNullParamName(){
100         String uriString = PATH + "?keyword=%E6%81%A4&page=";
101         assertEquals(PATH + "?keyword=%E6%81%A4&page=&=", addParameter(uriString, null, null, null));
102     }
103 
104     /**
105      * Test add parameter null param name 1.
106      */
107     @Test
108     public void testAddParameterNullParamName1(){
109         String uriString = PATH + "?keyword=%E6%81%A4&page=";
110         assertEquals(PATH + "?keyword=%E6%81%A4&page=&=2-5-8-12", addParameter(uriString, null, "2-5-8-12", null));
111     }
112 
113     /**
114      * Test add parameter null value.
115      */
116     @Test
117     public void testAddParameterNullValue(){
118         String uriString = PATH + "?keyword=%E6%81%A4&page=";
119         assertEquals(PATH + "?keyword=%E6%81%A4&page=&label=", addParameter(uriString, "label", null, null));
120     }
121 
122     //---------------------------------------------------------------
123 
124     /**
125      * Test add parameter null uri.
126      */
127     @Test
128     public void testAddParameterNullUri(){
129         assertEquals(EMPTY, addParameter(null, "label", "2-5-8-12", UTF8));
130     }
131 
132     /**
133      * Test add parameter empty uri.
134      */
135     @Test
136     public void testAddParameterEmptyUri(){
137         assertEquals(EMPTY, addParameter("", "label", "2-5-8-12", UTF8));
138     }
139 
140     /**
141      * Test add parameter blank uri.
142      */
143     @Test
144     public void testAddParameterBlankUri(){
145         assertEquals(EMPTY, addParameter(" ", "label", "2-5-8-12", UTF8));
146     }
147 }
