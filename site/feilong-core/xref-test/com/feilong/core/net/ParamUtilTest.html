112     @Test
113     public void testJoinValues(){
114         String value = "create_salesorder";
115         String value2 = "unionpay_mobile";
116 
117         Map<String, String> map = new HashMap<String, String>();
118         map.put("service", value);
119         map.put("paymentType", value2);
120 
121         assertEquals(EMPTY, ParamUtil.joinValuesOrderByIncludeKeys(map, "a", "b"));

123         assertEquals(value + value2, ParamUtil.joinValuesOrderByIncludeKeys(map, "service", "paymentType"));
124         assertEquals(value2 + value, ParamUtil.joinValuesOrderByIncludeKeys(map, "paymentType", "service"));
125     }
