33  /**
34   * {@link URL} 工具类.
35   * 
36   * <h3>URL组成部分:</h3>
37   * 
38   * <blockquote>
39   * <p>
40   * <img src="http://venusdrogon.github.io/feilong-platform/mysource/url-parts.png"/>
41   * </p>
42   * </blockquote>
43   * 
44   * <h3>URL 的长度上限</h3>
45   * 
46   * <blockquote>
47   * 
48   * <p>
49   * URL 的最大长度是多少？W3C 的 HTTP 协议 并没有限定; 然而经过试验,不同浏览器和 Web 服务器有不同的约定:
50   * </p>
51   * 
52   * <ul>
53   * <li><b>IE</b> 长度上限是 <b>2083</b> 字节,其中纯路径部分不能超过 <b>2048</b> 字节.</li>
54   * <li><b>Firefox</b> 地址栏中超过 <b>65536</b> 字符后就不再显示.</li>
55   * <li><b>Safari</b> 测试到 <b>80000</b> 字符还工作得好好的.</li>
56   * <li><b>Opera</b> 测试到 <b>190000</b> 字符还工作得好好的.</li>
57   * </ul>
58   * 
59   * <p>
60   * Web 服务器:
61   * </p>
62   * 
63   * <ul>
64   * <li><b>Apache Web</b> 服务器在接收到大约 <b>4000</b> 字符长的 URL 时候产生 <b>"413 Entity Too Large"</b> 错误.</li>
65   * <li><b>IIS</b> 默认接收的最大 URL 是 <b>16384</b> 字符.</li>
66   * </ul>
67   * </blockquote>
68   *
69   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
70   * @since 1.2.1
71   */

86      /**
87       * 将字符串 <code>spec</code> 转成 {@link URL}.
88       * 
89       * <h3>该方法特点:</h3>
90       * <blockquote>
91       * <ol>
92       * <li>处理 check exception,转成 uncheck exception</li>
93       * <li>如果使用new URL 出现异常,会尝试使用 {@link #toFileURL(String)}</li>
94       * </ol>
95       * </blockquote>
96       * 
97       * <h3>注意:</h3>
98       * 
99       * <blockquote>
100      * 
101      * <p>
102      * 如果直接调用 {@link URL#URL(String)} 方法,如果是文件路径, 比如
103      * </p>
104      * 
105      * <pre class="code">
106      * 
107      * String spec = "C:/Users/feilong/feilong/train/新员工/warmReminder/20160704141057.html";
108      * 
109      * URL newURL = new URL(spec);
110      * </pre>
111      * 
112      * 会抛出异常,
113      * 
114      * <pre class="code">
115      * Caused by: java.net.MalformedURLException: <span style="color:red"><b>unknown protocol: c</b></span>
116      * at java.net.URL.{@code <init>}(URL.java:593)
117      * at java.net.URL.{@code <init>}(URL.java:483)
118      * at java.net.URL.{@code <init>}(URL.java:432)
119      * ... 24 more
120      * </pre>
121      * 
122      * 此时 ,你需要修改成
123      * 
124      * <pre class="code">
125      * 
126      * String spec = "file://C:/Users/feilong/feilong/train/新员工/warmReminder/20160704141057.html";
127      * 
128      * URL newURL = new URL(spec);
129      * </pre>
130      * 
131      * 也就是说,此方法必须要有协议支持 (file URI scheme),如果你是文件需要转成 url的话,建议直接调用 {@link #toFileURL(String)}
132      * </blockquote>
133      * 
134      * @param spec
135      *            the <code>String</code> to parse as a URL.
136      * @return 如果 <code>spec</code> 是null,抛出 {@link NullPointerException}<br>
137      *         如果 <code>spec</code> 是blank,抛出 {@link IllegalArgumentException}<br>
138      * @see java.net.URL#URL(String)
139      * @see "org.apache.cxf.common.util.StringUtils#getURL(String)"
140      * @see "org.springframework.util.ResourceUtils#getURL(String)"
141      * @see <a href="https://en.wikipedia.org/wiki/File_URI_scheme">File_URI_scheme</a>
142      * @see "org.apache.xml.resolver.readers.TextCatalogReader#readCatalog(Catalog, String)"
143      * @see <a href="https://docs.oracle.com/javase/tutorial/networking/urls/creatingUrls.html">Creating a URL</a>
144      * @since 1.8.0
145      */
146     public static URL toURL(String spec){
147         Validate.notBlank(spec, "spec can't be blank!");
148 
149         //---------------------------------------------------------------
150         try{
151             return new URL(spec);
152         }catch (Exception e){
153             // no URL -> treat as file path
154             LOGGER.info("[new URL(\"{}\")] exception,cause by :[{}],will try call [toFileURL(\"{}\")]", spec, e.getMessage(), spec);
155             return toFileURL(spec);
156         }
157     }

161     /**
162      * 将字符串路径 <code>filePath</code> 转成{@link URL}.
163      * 
164      * <h3>示例:</h3>
165      * 
166      * <blockquote>
167      * 
168      * <pre class="code">
169      * 
170      * String spec = "C:\\Users\\feilong\\feilong\\train\\新员工\\warmReminder\\20160704141057.html";
171      * 
172      * URL newURL = toFileURL(spec);
173      * </pre>
174      * 
175      * </blockquote>
176      * 
177      * <h3>Using Java 7 你还可以使用:</h3>
178      * 
179      * <blockquote>
180      * Paths#get(string).toUri().toURL();
181      * 
182      * <p>
183      * However, you probably want to get a URI.<br>
184      * Eg, a URI begins with file:/// but a URL with file:/ (at least, that's what toString produces).
185      * </p>
186      * 
187      * 参见
188      * <a href="http://stackoverflow.com/questions/6098472/pass-a-local-file-in-to-url-in-java#21461738">pass-a-local-file-in-to-url-in-java
189      * </a>
190      * </blockquote>
191      *
192      * @param filePath
193      *            字符串路径
194      * @return the url
195      * @see java.io.File#toURI()
196      * @see java.net.URI#toURL()
197      * @see "org.apache.commons.io.FileUtils#toURLs(File[])"
198      * @since 1.8.0 change Access Modifiers
199      */
200     private static URL toFileURL(String filePath){
201         try{
202             return new File(filePath).toURI().toURL();// file.toURL() 已经过时,它不会自动转义 URL 中的非法字符
203         }catch (Exception e){
204             throw new URIParseException(format("filePath:[{}]", filePath), e);
205         }
206     }

210     /**
211      * 将 {@link URL}转成 {@link URI}.
212      *
213      * @param url
214      *            the url
215      * @return 如果 <code>url</code> 是null,返回null
216      * @see "org.springframework.util.ResourceUtils#toURI(URL)"
217      * @see java.net.URL#toURI()
218      * @since 1.2.2
219      * @since 1.8.7 如果 null==url,返回null
220      */
221     public static URI toURI(URL url){
222         if (null == url){
223             return null;
224         }
225         //---------------------------------------------------------------
226         try{
227             return url.toURI();
228         }catch (URISyntaxException e){
229             throw new URIParseException(format("input url:[{}]", url), e);
230         }
231     }

233     /**
234      * 打开当前<code>url</code>的连接,并且读取返回 <code>InputStream</code>.
235      * 
236      * <p>
237      * 这个方法是以下方法的简写
238      * </p>
239      * 
240      * <pre>
241      * url.openConnection().getInputStream()
242      * </pre>
243      * 
244      * @param url
245      *            the url
246      * @return 如果 <code>url</code> 是null,返回 null<br>
247      * @see java.net.URL#openStream()
248      * @since 1.9.2
249      */
250     public static InputStream openStream(URL url){
251         try{
252             return url == null ? null : url.openStream();
253         }catch (IOException e){
254             throw new UncheckedIOException(format("openStream url:[{}]", url), e);
255         }
256     }

260     /**
261      * 获取联合url,通过在指定的上下文中对给定的 spec 进行解析创建 URL,新的 URL 从给定的上下文 URL 和 spec 参数创建.
262      * 
263      * <p style="color:red">
264      * 网站地址拼接,请使用这个method
265      * </p>
266      * 
267      * <h3>示例:</h3>
268      * 
269      * <blockquote>
270      * 
271      * <pre class="code">
272      * 
273      * URL url = new URL("http://www.exiaoshuo.com/jinyiyexing/");
274      * URIUtil.getUnionUrl(url, "/jinyiyexing/1173348/")    =  http://www.exiaoshuo.com/jinyiyexing/1173348/
275      * </pre>
276      * 
277      * </blockquote>
278      * 
279      * @param context
280      *            要解析规范的上下文
281      * @param spec
282      *            the <code>String</code> to parse as a URL.
283      * @return 如果 <code>spec</code> 是null,抛出 {@link NullPointerException}<br>
284      *         如果 <code>spec</code> 是blank,抛出 {@link IllegalArgumentException}<br>
285      * @since 1.4.0
286      */
287     public static String getUnionUrl(URL context,String spec){
288         Validate.notBlank(spec, "spec can't be null!");
289 
290         //---------------------------------------------------------------
291         try{
292             return new URL(context, spec).toString();
293         }catch (Exception e){
294             String message = format("context:[{}],spec:[{}]", context, spec);
295             throw new URIParseException(message, e);
296         }
297     }
