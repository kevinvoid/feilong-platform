16  package com.feilong.core.bean;
17  
18  import static com.feilong.core.DatePattern.COMMON_DATE_AND_TIME;
19  import static com.feilong.core.bean.ConvertUtil.convert;
20  import static com.feilong.core.bean.ToStringConfig.DEFAULT_CONNECTOR;
21  
22  import java.math.BigDecimal;
23  import java.util.Calendar;
24  import java.util.Collection;
25  import java.util.Date;
26  
27  import org.apache.commons.collections4.Transformer;
28  
29  import com.feilong.core.lang.ClassUtil;
30  import com.feilong.core.util.transformer.DateToStringTransformer;
31  import com.feilong.core.util.transformer.NumberToStringTransformer;
32  
33  /**
34   * 处理值转换.
35   *
36   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
37   * @since 1.14.0
38   */
39  public final class ToStringHandler{
40  
41      /** 默认的 日期转字符串转换器. */
42      private static final Transformer<Date, String>   DEFAULT_DATETOSTRING_TRANSFORMER   = new DateToStringTransformer(COMMON_DATE_AND_TIME);
43  
44      /** 默认的 数字转字符串转换器. */
45      private static final Transformer<Number, String> DEFAULT_NUMBERTOSTRING_TRANSFORMER = new NumberToStringTransformer(
46                      com.feilong.core.NumberPattern.TWO_DECIMAL_POINTS);
47  
48      //---------------------------------------------------------------
49  
50      /** Don't let anyone instantiate this class. */
51      private ToStringHandler(){
52          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
53          //see 《Effective Java》 2nd
54          throw new AssertionError("No " + getClass().getName() + " instances for you!");
55      }
56  
57      //---------------------------------------------------------------
58  
59      /**
60       * To string value.
61       *
62       * @param toBeConvertedValue
63       *            the value
64       * @return 如果 <code>toBeConvertedValue</code> 是null,返回 null<br>
65       *         如果 <code>toBeConvertedValue</code> 是 {@link CharSequence},直接 toString返回<br>
66       *         如果 <code>toBeConvertedValue</code> 是 数组,那么调用 {@link ConvertUtil#toString(Object[], String)}<br>
67       *         如果 <code>toBeConvertedValue</code> 是 {@link Collection},那么调用 {@link ConvertUtil#toString(Object[], String)}<br>
68       *         如果 <code>toBeConvertedValue</code> 是 {@link Date},那么返回 {@link com.feilong.core.DatePattern#COMMON_DATE_AND_TIME} 格式字符串<br>
69       *         如果 <code>toBeConvertedValue</code> 是 {@link BigDecimal}或者是{@link Float}或者是 {@link Double},那么返回
70       *         {@link com.feilong.core.NumberPattern#TWO_DECIMAL_POINTS} 2 位小数点格式字符串<br>
71       *         其他调用 {@link com.feilong.core.bean.ConvertUtil#convert(Object, Class)}
72       */
73      static String toStringValue(Object toBeConvertedValue){
74          if (null == toBeConvertedValue){
75              return null;
76          }
77  
78          //---------------------------------------------------------------
79          //CharSequence
80          //since 1.14.0
81          if (ClassUtil.isInstance(toBeConvertedValue, CharSequence.class)){
82              return ((CharSequence) toBeConvertedValue).toString();
83          }
84  
85          //---------------------------------------------------------------
86  
87          //数组
88          //since 1.14.0
89          if (com.feilong.core.lang.ObjectUtil.isArray(toBeConvertedValue)){
90              return ConvertUtil.toString((Object[]) toBeConvertedValue, DEFAULT_CONNECTOR);
91          }
92  
93          //集合
94          //since 1.14.0
95          if (ClassUtil.isInstance(toBeConvertedValue, Collection.class)){
96              return ConvertUtil.toString((Collection<?>) toBeConvertedValue, DEFAULT_CONNECTOR);
97          }
98  
99          //---------------------------------------------------------------
100         //日期
101         if (ClassUtil.isInstance(toBeConvertedValue, Date.class)){
102             return DEFAULT_DATETOSTRING_TRANSFORMER.transform((Date) toBeConvertedValue);
103         }
104         //Calendar
105         if (ClassUtil.isInstance(toBeConvertedValue, Calendar.class)){
106             return DEFAULT_DATETOSTRING_TRANSFORMER.transform(((Calendar) toBeConvertedValue).getTime());
107         }
108 
109         //---------------------------------------------------------------
110         //数字
111         if (ClassUtil.isInstanceAnyClass(toBeConvertedValue, BigDecimal.class, Double.class, Float.class)){
112             return DEFAULT_NUMBERTOSTRING_TRANSFORMER.transform((Number) toBeConvertedValue);
113         }
114 
115         //---------------------------------------------------------------
116         return convert(toBeConvertedValue, String.class);
117     }
118 }
