65      /**
66       * 编译给定正则表达式 <code>regexPattern</code> ,并尝试将给定输入 <code>input</code> 与其匹配.
67       * 
68       * <p>
69       * {@link Pattern#matches(String, CharSequence)} 等价于{@link #getMatcher(String, CharSequence)}.matches();
70       * </p>
71       *
72       * @param regexPattern
73       *            正则表达式字符串,pls use {@link "RegexPattern"}
74       * @param input
75       *            The character sequence to be matched,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on
76       * @return 如果 <code>regexPattern</code> 是null,抛出 {@link NullPointerException}<br>
77       *         如果 <code>input</code> 是null,返回 false<br>
78       *         如果input 符合 regex的正则表达式格式,返回true,否则返回 false;<br>
79       * @see #getMatcher(String, CharSequence)
80       * @see Matcher#matches()
81       * @see Pattern#matches(String, CharSequence)
82       * @since 1.0.7
83       * @since 1.13.2 change, if input is null, return false from NPE
84       */
85      public static boolean matches(final String regexPattern,final CharSequence input){
86          Validate.notNull(regexPattern, "regexPattern can't be null!");
87          if (null == input){
88              return false;
89          }
90          return getMatcher(regexPattern, input).matches();
91      }

95      /**
96       * 返回在以前匹配操作期间由给定组捕获的输入子序列.
97       * 
98       * <p>
99       * 对于匹配器 m、输入序列 s 和组索引 g,表达式 m.group(g) 和 s.substring(m.start(g), m.end(g))是等效的.<br>
100      * 捕获组是从 1开始从左到右的索引.组0表示整个模式,因此表达式 m.group(0)等效于 m.group().
101      * </p>
102      * 
103      * <h3>示例:</h3>
104      * 
105      * <blockquote>
106      * 
107      * <pre class="code">
108      * String regexPattern = "(.*?)@(.*?)";
109      * String email = "feilong@163.com";
110      * 
111      * RegexUtil.group(regexPattern, email);
112      * </pre>
113      * 
114      * <b>返回:</b>
115      * 
116      * <pre class="code">
117      *    0 feilong@163.com
118      *    1 feilong
119      *    2 163.com
120      * </pre>
121      * 
122      * </blockquote>
123      * 
124      * @param regexPattern
125      *            正则表达式模式,比如 (.*?)@(.*?)
126      * @param input
127      *            需要被group的字符串,比如 feilong@163.com,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on
128      * @return 如果 <code>regexPattern</code> 是null,抛出 {@link NullPointerException}<br>
129      *         如果 <code>input</code> 是null,抛出 {@link NullPointerException}<br>
130      *         如果 匹配不了,返回 {@link java.util.Collections#emptyMap()}
131      * @see #getMatcher(String, CharSequence)
132      * @see Matcher#group(int)
133      * @since 1.0.7
134      */
135     public static Map<Integer, String> group(final String regexPattern,final CharSequence input){
136         Matcher matcher = getMatcher(regexPattern, input);
137         if (!matcher.matches()){
138             LOGGER.trace("[not matches] ,\n\tregexPattern:[{}] \n\tinput:[{}]", regexPattern, input);
139             return emptyMap();
140         }
141 
142         //---------------------------------------------------------------
143         int groupCount = matcher.groupCount();
144         Map<Integer, String> map = newLinkedHashMap(groupCount + 1);
145         for (int i = 0; i <= groupCount; ++i){
146             //匹配的索引
147             String groupValue = matcher.group(i); //map.put(0, matcher.group());// 捕获组是从 1 开始从左到右的索引.组0表示整个模式,因此表达式 m.group(0) 等效于 m.group().
148             LOGGER.trace("matcher group[{}],start-end:[{}-{}],groupValue:[{}]", i, matcher.start(i), matcher.end(i), groupValue);
149             map.put(i, groupValue);//groupValue
150         }
151 
152         //---------------------------------------------------------------
153 
154         if (LOGGER.isTraceEnabled()){
155             LOGGER.trace("regexPattern:[{}],input:[{}],groupMap:{}", regexPattern, input, map);
156         }
157         return map;
158     }

160     /**
161      * 返回在以前匹配操作期间由给定组捕获的输入子序列.
162      * 
163      * <p>
164      * 对于匹配器 m、输入序列 s 和组索引 g,表达式 m.group(g) 和 s.substring(m.start(g), m.end(g))是等效的.<br>
165      * 捕获组是从 1开始从左到右的索引.组0表示整个模式,因此表达式 m.group(0)等效于 m.group().
166      * </p>
167      * 
168      * <h3>示例:</h3>
169      * 
170      * <blockquote>
171      * 
172      * <pre class="code">
173      * 
174      * String regexPattern = "(.*?)@(.*?)";
175      * String email = "feilong@163.com";
176      * 
177      * RegexUtil.group(regexPattern, email, 1);<span style="color:green">//feilong</span>
178      * RegexUtil.group(regexPattern, email, 2);<span style="color:green">//163.com</span>
179      * 
180      * </pre>
181      * 
182      * </blockquote>
183      *
184      * @param regexPattern
185      *            正则表达式模式,比如 (.*?)@(.*?)
186      * @param input
187      *            需要被group的字符串,比如 feilong@163.com,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on
188      * @param groupNo
189      *            组号,从0开始
190      * @return 如果 <code>regexPattern</code> 是null,抛出 {@link NullPointerException}<br>
191      *         如果 <code>input</code> 是null,抛出 {@link NullPointerException}<br>
192      *         如果 {@code input < 0} ,抛出 {@link IllegalArgumentException}<br>
193      * @see #getMatcher(String, CharSequence)
194      * @see Matcher#group(int)
195      * @since 1.0.7
196      */
197     public static String group(final String regexPattern,final CharSequence input,final int groupNo){
198         Validate.isTrue(groupNo >= 0, "groupNo must >=0");
199 
200         Map<Integer, String> map = group(regexPattern, input);
201         return map.get(groupNo);
202     }

207      * Gets the matcher.
208      *
209      * @param regexPattern
210      *            正则表达式字符串,pls use {@link "RegexPattern"}
211      * @param input
212      *            The character sequence to be matched,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on

214      *         如果 <code>input</code> 是null,抛出 {@link NullPointerException}<br>
215      * @see Pattern#compile(String)
216      * @since 1.0.7

218     private static Matcher getMatcher(final String regexPattern,final CharSequence input){
219         return getMatcher(regexPattern, input, 0);
220     }

222     /**
223      * Gets the matcher.
224      *
225      * @param regexPattern
226      *            正则表达式字符串,pls use {@link "RegexPattern"}
227      * @param input
228      *            The character sequence to be matched,support {@link String},{@link StringBuffer},{@link StringBuilder}... and so on
229      * @param flags
230      *            如果需要多个组合,可以使用罗辑或
231      *            <blockquote><code>Pattern.compile(regex, CASE_INSENSITIVE | DOTALL);</code></blockquote>
232      *            <ul>
233      *            <li>{@link Pattern#CASE_INSENSITIVE} 匹配字符时与大小写无关,该标志默认只考虑US ASCII字符.</li>
234      *            <li>{@link Pattern#MULTILINE} ^和$匹配一行的开始和结尾,而不是整个输入</li>
235      *            <li>{@link Pattern#UNICODE_CASE} 当与CASE_INSENSITIVE结合时,使用Unicode字母匹配</li>
236      *            <li>{@link Pattern#CANON_EQ} 考虑Unicode字符的规范等价</li>
237      *            <li>{@link Pattern#DOTALL} 当使用此标志时,.符号匹配包括行终止符在内的所有字符</li>
238      *            <li>{@link Pattern#UNIX_LINES} 当在多行模式下匹配^和$时,只将'\n'看作行终止符</li>
239      *            <li>{@link Pattern#LITERAL} 启用模式的字面值解析.</li>
240      *            <li>{@link Pattern#COMMENTS} 模式中允许空白和注释. <br>
241      *            此模式将忽略空白和在结束行之前以 # 开头的嵌入式注释. <br>
242      *            通过嵌入式标志表达式 (?x) 也可以启用注释模式. <br>
243      *            </li>
244      *            </ul>
245      * @return 如果 <code>regexPattern</code> 是null,抛出 {@link NullPointerException}<br>
246      *         如果 <code>input</code> 是null,抛出 {@link NullPointerException}<br>
247      * @see Pattern#compile(String, int)
248      * @since 1.5.3
249      */
250     private static Matcher getMatcher(final String regexPattern,final CharSequence input,final int flags){
251         Validate.notNull(regexPattern, "regexPattern can't be null!");
252         Validate.notNull(input, "input can't be null!");
253 
254         Pattern pattern = buildPattern(regexPattern, flags);
255         return pattern.matcher(input);
256     }

260     /**
261      * 如果cache中有,那么直接返回, 如果没有构造一个并塞到cache中.
262      *
263      * @param regexPattern
264      *            the regex pattern
265      * @param flags
266      *            the flags
267      * @return the pattern
268      * @since 1.10.6
269      */
270     public static Pattern buildPattern(final String regexPattern,final int flags){
271         final String key = regexPattern + "@" + flags;
272         Pattern pattern = PATTERN_CACHE.get(key);
273 
274         if (null == pattern){
275             pattern = Pattern.compile(regexPattern, flags);
276             PATTERN_CACHE.put(key, pattern);
277         }
278         return pattern;
279     }
