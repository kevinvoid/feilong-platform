16  package com.feilong.core.util.closure;
17  
18  import org.apache.commons.collections4.Closure;
19  import org.apache.commons.lang3.Validate;
20  
21  import com.feilong.core.bean.PropertyUtil;
22  
23  /**
24   * {@link org.apache.commons.collections4.Closure} 实现,用来更新指定属性的指定值.
25   * <h3>构造函数:</h3>
26   * 
27   * <blockquote>
28   * 
29   * BeanPropertyValueChangeClosure 构造函数提供两个参数, 属性名称和指定的属性值
30   * 
31   * <pre class="code">
32   * public BeanPropertyValueChangeClosure(String propertyName, Object propertyValue)
33   * </pre>
34   * 
35   * <b>注意:</b> Possibly indexed and/or nested name of the property to be
36   * modified,参见<a href="../../bean/BeanUtil.html#propertyName">propertyName</a>.
37   * </blockquote>
38   * 
39   * <h3>典型的使用示例:</h3>
40   * 
41   * <blockquote>
42   * 
43   * <pre class="code">
44   * // create the closure
45   * BeanPropertyValueChangeClosure closure = new BeanPropertyValueChangeClosure("activeEmployee", Boolean.TRUE);
46   *
47   * // update the Collection
48   * CollectionUtils.forAllDo(peopleCollection, closure);
49   * </pre>
50   * 
51   * 上面的示例将会提取 <code>peopleCollection</code> 的每个 person 对象, 并且更新<code>activeEmployee</code> 属性值为<code>true</code>.
52   * 
53   * </blockquote>
54   * 
55   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
56   * @param <T>
57   *            the generic type
58   * @see org.apache.commons.beanutils.BeanPropertyValueChangeClosure
59   * @see org.apache.commons.collections4.CollectionUtils#forAllDo(Iterable, Closure)
60   * @see org.apache.commons.collections4.IterableUtils#forEach(Iterable, Closure)
61   * @since 1.10.2
62   */
63  public class BeanPropertyValueChangeClosure<T> implements Closure<T>{
64  
65      /**
66       * 指定bean对象排序属性名字.
67       * 
68       * <p>
69       * 泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
70       * <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>
71       * </p>
72       */
73      private final String propertyName;
74  
75      /** 指定的属性值. */
76      private final Object propertyValue;
77  
78      //---------------------------------------------------------------
79  
80      /**
81       * Instantiates a new bean property value change closure.
82       * 
83       * <p>
84       * 如果 <code>propertyName</code> 是null,抛出 {@link NullPointerException}<br>
85       * 如果 <code>propertyName</code> 是blank,抛出 {@link IllegalArgumentException}<br>
86       * </p>
87       *
88       * @param propertyName
89       *            指定bean对象排序属性名字.
90       *            <p>
91       *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
92       *            <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>
93       *            </p>
94       * @param propertyValue
95       *            the value
96       */
97      public BeanPropertyValueChangeClosure(String propertyName, Object propertyValue){
98          Validate.notBlank(propertyName, "propertyName can't be blank!");
99          this.propertyName = propertyName;
100         this.propertyValue = propertyValue;
101     }
102 
103     //---------------------------------------------------------------
104 
105     /*
106      * (non-Javadoc)
107      * 
108      * @see org.apache.commons.collections4.Closure#execute(java.lang.Object)
109      */
110     @Override
111     public void execute(T input){
112         if (null == input){//如果 input是null,跳过去
113             return;
114         }
115 
116         PropertyUtil.setProperty(input, propertyName, propertyValue);
117     }
118 }
