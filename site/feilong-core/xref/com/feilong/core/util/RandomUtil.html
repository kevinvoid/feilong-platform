18  import java.util.Random;

20  import org.apache.commons.lang3.RandomStringUtils;
21  import org.apache.commons.lang3.Validate;

23  /**
24   * 随机数工具类.
25   * 
26   * <ul>
27   * <li>{@link java.lang.Math#random()}底层也是调用的 new Random(),值＝Random nextDouble()</li>
28   * <li>把 {@link java.util.Random}对象作为一个全局实例(static)来使用.Java中 {@link java.util.Random} 是线程安全的(内部进行了加锁处理);</li>
29   * <li>伪随机数</li>
30   * <li>生成随机数的算法有很多种,最简单也是最常用的就是 "线性同余法":第n+1个数=(第n个数*29+37) % 1000,其中%是"求余数"运算符.</li>
31   * </ul>
32   *
33   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
34   * @see java.lang.Math#random()
35   * @see org.apache.commons.lang3.RandomUtils
36   * @see org.apache.commons.lang3.RandomStringUtils
37   * @since 1.0.0
38   */
39  public final class RandomUtil{
40  
41      /**
42       * Random object used by random method.
43       * 
44       * <p>
45       * 为了确保同一毫秒不能返回相同的值,不同声明在方法里面.<br>
46       * 把Random对象作为一个全局实例(static)来使用. Java中Random是线程安全的(内部进行了加锁处理);
47       * </p>
48       * 
49       * @see org.apache.commons.lang3.RandomUtils
50       * @since 1.0.7
51       */
52      private static final Random JVM_RANDOM = new Random();
53  
54      /** Don't let anyone instantiate this class. */
55      private RandomUtil(){
56          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
57          //see 《Effective Java》 2nd
58          throw new AssertionError("No " + getClass().getName() + " instances for you!");
59      }

61      //---------------------------------------------------------------
62  
63      /**
64       * 生成一个指定长度<code>length</code>的 <b>随机正整数</b>.
65       * 
66       * <h3>示例:</h3>
67       * <blockquote>
68       * 
69       * <pre class="code">
70       * RandomUtil.createRandomWithLength(2)
71       * 生成的结果是可能是 89
72       * </pre>
73       * 
74       * </blockquote>
75       *
76       * @param length
77       *            设定所取出随机数的长度.
78       * @return 如果 <code>length</code> {@code <=0} ,抛出 {@link IllegalArgumentException}
79       */
80      public static long createRandomWithLength(int length){
81          Validate.isTrue(length > 0, "input param [length] must >0,but is [%s]", length);
82          long num = 1;
83          for (int i = 0; i < length; ++i){
84              num = num * 10;
85          }
86  
87          // 该值大于等于 0.0 且小于 1.0 正号的 double 值
88          double random = JVM_RANDOM.nextDouble();
89          random = random < 0.1 ? random + 0.1 : random;// 可能出现 0.09346924349151808
90          return (long) (random * num);
91      }

95      /**
96       * 随机抽取字符串<code>char</code>,拼接成指定长度<code>length</code>的字符串.
97       * 
98       * <h3>说明:</h3>
99       * <blockquote>
100      * <ol>
101      * <li>常用于生成验证码</li>
102      * </ol>
103      * </blockquote>
104      * 
105      * <h3>示例:</h3>
106      * <blockquote>
107      * 
108      * <pre class="code">
109      * RandomUtil.createRandomFromString({@link com.feilong.core.Alphabet#DECIMAL_AND_LETTERS Alphabet.DECIMAL_AND_LETTERS}, 5)
110      * 
111      * 从 {@link com.feilong.core.Alphabet#DECIMAL_AND_LETTERS  Alphabet.DECIMAL_AND_LETTERS} 随机抽取字符,组成长度是5的字符串
112      * 生成的结果是可能是IFSMB
113      * </pre>
114      * 
115      * </blockquote>
116      * 
117      * @param str
118      *            被抽取的字符串,比如{@link com.feilong.core.Alphabet#DECIMAL_AND_LETTERS}
119      * @param length
120      *            指定字符串长度,比如 5
121      * @return 如果 <code>str</code> 是null,抛出 {@link NullPointerException}<br>
122      *         如果 <code>str</code> 是blank,抛出 {@link IllegalArgumentException}<br>
123      *         如果 <code>length</code> {@code <=0}, 抛出 {@link IllegalArgumentException}
124      * 
125      * @see org.apache.commons.lang3.RandomStringUtils#random(int, String)
126      */
127     public static String createRandomFromString(String str,int length){
128         Validate.notBlank(str, "str can't be null/empty!");
129         Validate.isTrue(length > 0, "input param [length] must >0,but is [%s]", length);
130         return RandomStringUtils.random(length, str);
131     }
