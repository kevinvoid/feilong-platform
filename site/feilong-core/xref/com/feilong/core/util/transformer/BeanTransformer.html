16  package com.feilong.core.util.transformer;
17  
18  import java.io.Serializable;
19  
20  import org.apache.commons.collections4.Transformer;
21  import org.apache.commons.lang3.Validate;
22  
23  import com.feilong.core.bean.PropertyUtil;
24  import com.feilong.core.lang.reflect.ConstructorUtil;
25  
26  /**
27   * 将传入的bean转成指定类型 <code>toBeanType</code> 的Bean 的转换器.
28   * 
29   * <p>
30   * 如果只需要转换特定的属性(忽略其他属性),此时可以使用 <code>includePropertyNames</code> 属性;<br>
31   * 如果没有传递此属性,那么默认会转换所有的属性
32   * </p>
33   *
34   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
35   * @param <I>
36   *            输入类型
37   * @param <O>
38   *            输出类型
39   * @see org.apache.commons.beanutils.BeanToPropertyValueTransformer
40   * @since 1.10.1
41   */
42  public class BeanTransformer<I, O> implements Transformer<I, O>,Serializable{
43  
44      /** The Constant serialVersionUID. */
45      private static final long serialVersionUID = -5873559262715038376L;
46  
47      /**
48       * 转成Bean的类型.
49       */
50      private final Class<O>    toBeanType;
51  
52      /**
53       * 指定转换属性的名字(如果).
54       */
55      private final String[]    includePropertyNames;
56  
57      //---------------------------------------------------------------
58  
59      /**
60       * Instantiates a new bean transformer.
61       * 
62       * <p>
63       * 如果 <code>toBeanType</code> 是null,抛出 {@link NullPointerException}<br>
64       * </p>
65       *
66       * @param toBeanType
67       *            转成Bean的类型.
68       * @param includePropertyNames
69       *            包含的属性数组名字数组,(can be nested/indexed/mapped/combo),如果是null或者empty,那么复制所有的属性
70       */
71      public BeanTransformer(Class<O> toBeanType, String...includePropertyNames){
72          Validate.notNull(toBeanType, "toBeanType can't be null!");
73          this.toBeanType = toBeanType;
74          this.includePropertyNames = includePropertyNames;
75      }
76  
77      //---------------------------------------------------------------
78  
79      /*
80       * (non-Javadoc)
81       * 
82       * @see org.apache.commons.collections4.Transformer#transform(java.lang.Object)
83       */
84      @Override
85      public O transform(I inputBean){
86          if (null == inputBean){
87              return null;
88          }
89  
90          O outputBean = ConstructorUtil.newInstance(toBeanType);
91  
92          PropertyUtil.copyProperties(outputBean, inputBean, includePropertyNames);
93          return outputBean;
94      }
95  }
