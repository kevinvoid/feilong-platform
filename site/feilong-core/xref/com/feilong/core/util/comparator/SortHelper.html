28   * 排序的辅助类.

33  public final class SortHelper{
34  
35      /** 正序排序因子. */
36      private static final String ASC  = "ASC";
37  
38      /** 倒序排序因子. */
39      private static final String DESC = "DESC";
40  
41      //---------------------------------------------------------------
42  
43      /** Don't let anyone instantiate this class. */
44      private SortHelper(){
45          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
46          //see 《Effective Java》 2nd
47          throw new AssertionError("No " + getClass().getName() + " instances for you!");
48      }

52      /**
53       * 将 <code>propertyNameAndOrder</code>字符串转换成数组.
54       * 
55       * <p>
56       * 会先进行trim操作,再使用{@link StringUtils#SPACE} 分隔成数组
57       * </p>
58       * 
59       * <h3>示例:</h3>
60       * 
61       * <blockquote>
62       * 
63       * <pre class="code">
64       * SortHelper.parsePropertyNameAndOrder("name") = toArray("name", null);
65       * SortHelper.parsePropertyNameAndOrder("name ") = toArray("name", null);
66       * SortHelper.parsePropertyNameAndOrder(" name") = toArray("name", null);
67       * 
68       * SortHelper.parsePropertyNameAndOrder("name asc") = toArray("name", "aSc");
69       * SortHelper.parsePropertyNameAndOrder("name aSc") = toArray("name", "asc");
70       * SortHelper.parsePropertyNameAndOrder("name   aSc") = toArray("name", "asc");
71       * 
72       * SortHelper.parsePropertyNameAndOrder("name desc") = toArray("name", "desc");
73       * 
74       * SortHelper.parsePropertyNameAndOrder("name deSc") = toArray("name", "deSc");
75       * </pre>
76       * 
77       * </blockquote>
78       * 
79       * @param propertyNameAndOrder
80       *            属性名称和排序因子,
81       * 
82       *            <p>
83       *            格式可以是纯的属性名称, 比如 "name"; 也可以是属性名称+排序因子(以空格分隔),比如 "name desc"
84       *            </p>
85       * 
86       *            <h3>说明:</h3>
87       *            <blockquote>
88       * 
89       *            <dl>
90       *            <dt>关于属性名称</dt>
91       *            <dd>
92       *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be
93       *            modified,参见<a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,<br>
94       *            该属性对应的value 必须实现 {@link Comparable}接口.
95       *            </dd>
96       * 
97       *            <dt>关于排序因子</dt>
98       *            <dd>
99       *            可以没有排序因子<br>
100      *            如果有,值可以是asc(顺序),desc(倒序)两种;<br>
101      *            如果没有,默认按照asc(顺序)排序;<br>
102      *            此外,asc/desc忽略大小写
103      *            </dd>
104      * 
105      *            </dl>
106      * 
107      *            </blockquote>
108      * @return 属性名称和排序因子的数组
109      * @throws NullPointerException
110      *             如果 <code>propertyNameAndOrder</code> 是null
111      * @throws IllegalArgumentException
112      *             如果 <code>propertyNameAndOrder</code> 是blank;<br>
113      *             或者如果<code>propertyNameAndOrder</code>使用{@link StringUtils#SPACE}转换的数组length {@code >} 2<br>
114      *             或者如果<code>propertyNameAndOrder</code>使用{@link StringUtils#SPACE}转换的数组length {@code =} 2,但是第二个值不是asc也不是desc
115      * @see com.feilong.core.lang.StringUtil#tokenizeToStringArray(String, String)
116      */
117     public static String[] parsePropertyNameAndOrder(String propertyNameAndOrder){
118         Validate.notBlank(propertyNameAndOrder, "propertyNameAndOrder can't be blank!");
119 
120         //---------------------------------------------------------------
121         //空格
122         String[] array = StringUtil.tokenizeToStringArray(propertyNameAndOrder.trim(), SPACE);
123         if (array.length > 2){
124             String message = format("propertyNameAndOrder:[{}] has more than 1 space,must max 1 space", propertyNameAndOrder);
125             throw new IllegalArgumentException(message);
126         }
127 
128         //---------------------------------------------
129         String order = null;
130         if (array.length == 2){
131             order = array[1];//排序因子
132             if (!(order.equalsIgnoreCase(ASC) || order.equalsIgnoreCase(DESC))){
133                 String pattern = "propertyNameAndOrder:[{}] 's order:[{}] must ignoreCase equals [asc or desc]";
134                 throw new IllegalArgumentException(format(pattern, propertyNameAndOrder, order));
135             }
136         }
137 
138         //---------------------------------------------
139         return toArray(array[0], order);
140     }

144     /**
145      * 判断属性名称和排序因子数组是不是asc排序.
146      * 
147      * <h3>示例:</h3>
148      * 
149      * <blockquote>
150      * 
151      * <pre class="code">
152      * SortHelper.isAsc(toArray("name", null)) = true;
153      * 
154      * SortHelper.isAsc(toArray("name", "asC")) = true;
155      * SortHelper.isAsc(toArray("name", "AsC")) = true;
156      * SortHelper.isAsc(toArray("name", "ASC")) = true;
157      * 
158      * SortHelper.isAsc(toArray("name", "dEsc")) = false;
159      * SortHelper.isAsc(toArray("name", "desc")) = false;
160      * </pre>
161      * 
162      * </blockquote>
163      * 
164      * @param propertyNameAndOrderArray
165      *            属性名称和排序因子数组,length 是2<br>
166      *            第一个元素是属性名称,比如 "name",第二个元素是排序因子,比如asc或者是desc或者是null<br>
167      * @return 如果数组第二个值是null或者empty,或者值是asc(忽略大小写),那么判定是asc排序
168      * @throws NullPointerException
169      *             如果 <code>propertyNameAndOrderArray</code> 是null
170      * @throws IllegalArgumentException
171      *             如果 <code>propertyNameAndOrderArray</code> 是empty<br>
172      *             或者<code>propertyNameAndOrderArray</code>的length不是2<br>
173      *             或者解析出来的order值不是[null/ASC/DESC] 中的任意一个
174      */
175     public static boolean isAsc(String[] propertyNameAndOrderArray){
176         Validate.notEmpty(propertyNameAndOrderArray, "propertyNameAndOrderArray can't be null/empty!");
177 
178         Validate.isTrue(
179                         2 == propertyNameAndOrderArray.length,
180                         "propertyNameAndOrderArray.length must 2, but length is:[%s],propertyNameAndOrderArray:[%s]",
181                         propertyNameAndOrderArray.length,
182                         propertyNameAndOrderArray);
183 
184         //---------------------------------------------------------------
185 
186         String order = propertyNameAndOrderArray[1];
187 
188         Validate.isTrue(
189                         null == order || ASC.equalsIgnoreCase(order) || DESC.equalsIgnoreCase(order),
190                         "order value must one of [null/ASC/DESC], but is:[%s],propertyNameAndOrderArray:[%s]",
191                         order,
192                         propertyNameAndOrderArray);
193 
194         //---------------------------------------------------------------
195 
196         return null == order || ASC.equalsIgnoreCase(order);
197     }
