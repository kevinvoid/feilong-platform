52      /**
53       * 按照不同指定属性 <code>propertyNameAndOrders</code> 排序的 {@link Comparator}.
54       *
55       * @param <T>
56       *            the generic type
57       * @param propertyNameAndOrders
58       *            属性名称和排序因子,
59       * 
60       *            <p>
61       *            格式可以是纯的属性名称, 比如 "name"; 也可以是属性名称+排序因子(以空格分隔),比如 "name desc"
62       *            </p>
63       * 
64       *            <h3>说明:</h3>
65       *            <blockquote>
66       * 
67       *            <dl>
68       *            <dt>关于属性名称</dt>
69       *            <dd>
70       *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be
71       *            modified,参见<a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,<br>
72       *            该属性对应的value 必须实现 {@link Comparable}接口.
73       *            </dd>
74       * 
75       *            <dt>关于排序因子</dt>
76       *            <dd>
77       *            可以没有排序因子<br>
78       *            如果有,值可以是asc(顺序),desc(倒序)两种;<br>
79       *            如果没有,默认按照asc(顺序)排序;<br>
80       *            此外,asc/desc忽略大小写
81       *            </dd>
82       * 
83       *            </dl>
84       * 
85       *            </blockquote>
86       * @return 如果propertyNameAndOrders是单值,那么直接调用 {@link #propertyComparator(String)} 返回
87       * @throws NullPointerException
88       *             如果 <code>propertyNameAndOrders</code> 是null,<br>
89       *             或者有元素是 null;
90       * @throws IllegalArgumentException
91       *             如果 <code>propertyNameAndOrders</code> 是empty,<br>
92       *             或者有元素是 blank
93       * @see org.apache.commons.collections4.ComparatorUtils#chainedComparator(java.util.Collection)
94       * @since 1.10.2 support propertyNameAndOrder
95       */
96      public static <T> Comparator<T> chainedComparator(String...propertyNameAndOrders){
97          Validate.notEmpty(propertyNameAndOrders, "propertyNameAndOrders can't be null/empty!");
98  
99          //如果propertyNameAndOrders是单值,那么直接调用 com.feilong.core.util.comparator.BeanComparatorUtil.propertyComparator(String) 返回
100         if (1 == propertyNameAndOrders.length){
101             return propertyComparator(propertyNameAndOrders[0]);
102         }
103 
104         //---------------------------------------------------------------
105 
106         List<Comparator<T>> comparators = newArrayList();
107         for (String propertyNameAndOrder : propertyNameAndOrders){
108             Validate.notBlank(propertyNameAndOrder, "propertyNameAndOrder can't be blank!");
109 
110             String[] propertyNameAndOrderArray = SortHelper.parsePropertyNameAndOrder(propertyNameAndOrder);
111 
112             //注意:此处不要使用 propertyComparator(propertyName)
113 
114             //因为,PropertyComparator 如果属性值相同,会使用其他规则继续比较(为了TreeMap/treeSet), 
115             //也就是说,通常而言一次就比较出顺序,后续的propertyNameAndOrders 就没作用了
116             Comparator instance = ComparatorUtils.nullHighComparator(ComparableComparator.comparableComparator()); //null排在最后面  
117 
118             BeanComparator<T> beanComparator = new BeanComparator<>(propertyNameAndOrderArray[0], instance);
119             comparators.add(isAsc(propertyNameAndOrderArray) ? beanComparator : reversedComparator(beanComparator));
120         }
121         return ComparatorUtils.chainedComparator(comparators);
122     }

126     /**
127      * 指定属性 <code>propertyName</code> ,按照自然顺序 排序的 {@link Comparator}.
128      *
129      * @param <T>
130      *            the generic type
131      * @param propertyNameAndOrder
132      *            属性名称和排序因子,
133      * 
134      *            <p>
135      *            格式可以是纯的属性名称, 比如 "name"; 也可以是属性名称+排序因子(以空格分隔),比如 "name desc"
136      *            </p>
137      * 
138      *            <h3>说明:</h3>
139      *            <blockquote>
140      * 
141      *            <dl>
142      *            <dt>关于属性名称</dt>
143      *            <dd>
144      *            泛型T对象的属性名称,Possibly indexed and/or nested name of the property to be
145      *            modified,参见<a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,<br>
146      *            该属性对应的value 必须实现 {@link Comparable}接口.
147      *            </dd>
148      * 
149      *            <dt>关于排序因子</dt>
150      *            <dd>
151      *            可以没有排序因子<br>
152      *            如果有,值可以是asc(顺序),desc(倒序)两种;<br>
153      *            如果没有,默认按照asc(顺序)排序;<br>
154      *            此外,asc/desc忽略大小写
155      *            </dd>
156      * 
157      *            </dl>
158      * 
159      *            </blockquote>
160      * @return {@link PropertyComparator}
161      * @throws NullPointerException
162      *             如果 <code>propertyNameAndOrder</code> 是null
163      * @throws IllegalArgumentException
164      *             如果 <code>propertyNameAndOrder</code> 是blank
165      * @see PropertyComparator#PropertyComparator(String)
166      */
167     public static <T> Comparator<T> propertyComparator(String propertyNameAndOrder){
168         Validate.notBlank(propertyNameAndOrder, "propertyNameAndOrder can't be blank!");
169 
170         String[] propertyNameAndOrderArray = SortHelper.parsePropertyNameAndOrder(propertyNameAndOrder);
171 
172         PropertyComparator<T> propertyComparator = new PropertyComparator<>(propertyNameAndOrderArray[0]);
173         return isAsc(propertyNameAndOrderArray) ? propertyComparator : reversedComparator(propertyComparator);
174     }

178     /**
179      * 指定属性 <code>propertyName</code> 按照固定顺序值 <code>propertyValues</code> 排序的 {@link Comparator}.
180      *
181      * @param <T>
182      *            the generic type
183      * @param <V>
184      *            the value type
185      * @param propertyName
186      *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
187      *            <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,该属性对应的value 必须实现 {@link Comparable}接口.
188      * @param propertyValues
189      *            the property values
190      * @return the comparator
191      * @throws NullPointerException
192      *             如果 <code>propertyName</code> 是null,或者<code>propertyValues</code> 是null
193      * @throws IllegalArgumentException
194      *             如果 <code>propertyName</code> 是blank
195      * @see #propertyComparator(String, List)
196      */
197     @SafeVarargs
198     public static <T, V> Comparator<T> propertyComparator(String propertyName,V...propertyValues){
199         Validate.notBlank(propertyName, "propertyName can't be blank!");
200         Validate.notNull(propertyValues, "propertyValues can't be null!");
201         return propertyComparator(propertyName, toList(propertyValues));
202     }

204     /**
205      * 指定属性 <code>propertyName</code> 按照固定顺序值 <code>propertyValues</code> 排序的 {@link Comparator}.
206      * 
207      * <p>
208      * 调用 {@link #propertyComparator(String, List, UnknownObjectBehavior)},默认 {@link UnknownObjectBehavior#AFTER}
209      * </p>
210      *
211      * @param <T>
212      *            the generic type
213      * @param <V>
214      *            the value type
215      * @param propertyName
216      *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
217      *            <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,该属性对应的value 必须实现 {@link Comparable}接口.
218      * @param propertyValues
219      *            the property values
220      * @return the comparator
221      * @throws NullPointerException
222      *             如果 <code>propertyName</code> 是null,或者<code>propertyValues</code> 是null
223      * @throws IllegalArgumentException
224      *             如果 <code>propertyName</code> 是blank
225      * @see #propertyComparator(String, List, UnknownObjectBehavior)
226      */
227     public static <T, V> Comparator<T> propertyComparator(String propertyName,List<V> propertyValues){
228         Validate.notBlank(propertyName, "propertyName can't be blank!");
229         Validate.notNull(propertyValues, "propertyValues can't be null!");
230         return propertyComparator(propertyName, propertyValues, UnknownObjectBehavior.AFTER);
231     }

233     /**
234      * 指定属性 <code>propertyName</code> 按照固定顺序值 <code>propertyValues</code>,并且可以指定 <code>unknownObjectBehavior</code> 排序的 {@link Comparator}.
235      *
236      * @param <T>
237      *            the generic type
238      * @param <V>
239      *            the value type
240      * @param propertyName
241      *            泛型T对象指定的属性名称,Possibly indexed and/or nested name of the property to be modified,参见
242      *            <a href="../../bean/BeanUtil.html#propertyName">propertyName</a>,该属性对应的value 必须实现 {@link Comparable}接口.
243      * @param propertyValues
244      *            the property values
245      * @param unknownObjectBehavior
246      *            the unknown object behavior
247      * @return the comparator
248      * @throws NullPointerException
249      *             如果 <code>propertyName</code> 是null,或者<code>propertyValues</code> 是null
250      * @throws IllegalArgumentException
251      *             如果 <code>propertyName</code> 是blank
252      * @see PropertyComparator#PropertyComparator(String, Comparator)
253      * @see FixedOrderComparator#FixedOrderComparator(List)
254      * @since 1.8.2
255      */
256     public static <T, V> Comparator<T> propertyComparator(
257                     String propertyName,
258                     List<V> propertyValues,
259                     UnknownObjectBehavior unknownObjectBehavior){
260         Validate.notBlank(propertyName, "propertyName can't be blank!");
261         Validate.notNull(propertyValues, "propertyValues can't be null!");
262         FixedOrderComparator<V> fixedOrderComparator = ComparatorUtil.buildFixedOrderComparator(propertyValues, unknownObjectBehavior);
263         return new PropertyComparator<>(propertyName, fixedOrderComparator);
264     }
