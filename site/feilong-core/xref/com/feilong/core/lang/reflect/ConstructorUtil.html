16  package com.feilong.core.lang.reflect;
17  
18  import java.lang.reflect.Constructor;
19  
20  import org.apache.commons.lang3.Validate;
21  import org.apache.commons.lang3.reflect.ConstructorUtils;
22  
23  import com.feilong.core.lang.ClassUtil;
24  import com.feilong.tools.slf4j.Slf4jUtil;
25  
26  /**
27   * 使用反射方法请求构造函数创建新实例的工具类,可以简化程序中使用反射方式创建对象的代码,focused on constructors.
28   * 
29   * <h3>方法介绍:</h3>
30   * 
31   * <blockquote>
32   * <ol>
33   * <li>{@link #newInstance(Class, Object...)} 获得实例</li>
34   * <li>{@link #newInstance(Class, Object[], Class[])} 获得实例</li>
35   * </ol>
36   * 
37   * 上面三个方法,底层均调用了 {@link ConstructorUtils#invokeConstructor(Class, Object[], Class[])} ,此方法 会
38   * {@link ConstructorUtils#getMatchingAccessibleConstructor(Class, Class...)} 自动根据类型活动最匹配的构造函数,并且将异常转成了 {@link ReflectException} 以便调用的时候使用
39   * 
40   * <p>
41   * 因此,不用担心 int{@code -->}Integer 原始类型参数和包装类型参数,以及父类/子类 这样的参数不匹配带来的问题
42   * </p>
43   * 
44   * </blockquote>
45   * 
46   * <h3>如果不想使用自动匹配的特性</h3>
47   * 
48   * <blockquote>
49   * 
50   * <p>
51   * 您可以使用原生方法,下面两个是 获得精准的构造函数并实例,如果参数类型不匹配,那么就会抛异常
52   * </p>
53   * 
54   * <ol>
55   * <li>{@link ConstructorUtils#invokeExactConstructor(Class, Object...)} 获得实例</li>
56   * <li>{@link ConstructorUtils#invokeExactConstructor(Class, Object[], Class[])} 获得实例</li>
57   * </ol>
58   * 
59   * 下面还有三个方法是,获得构造函数而不实例
60   * <ol>
61   * <li>{@link ConstructorUtils#getAccessibleConstructor(Constructor)}</li>
62   * <li>{@link ConstructorUtils#getAccessibleConstructor(Class, Class...)}</li>
63   * <li>{@link ConstructorUtils#getMatchingAccessibleConstructor(Class, Class...)}</li>
64   * </ol>
65   * 
66   * </blockquote>
67   * 
68   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
69   * @see org.apache.commons.lang3.reflect.ConstructorUtils
70   * @see "org.springframework.beans.BeanUtils.instantiateClass"
71   * @since 1.0.7
72   */
73  public final class ConstructorUtil{
74  
75      /** Don't let anyone instantiate this class. */
76      private ConstructorUtil(){
77          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
78          //see 《Effective Java》 2nd
79          throw new AssertionError("No " + getClass().getName() + " instances for you!");
80      }
81  
82      //---------------------------------------------------------------
83  
84      // [start] newInstance
85  
86      /**
87       * 新建实例,返回指定类型 <code>klass</code> 的实例,使用正确的构造函数使用参数类型<code>parameterValues</code>.
88       * 
89       * <h3>示例:</h3>
90       * 
91       * <blockquote>
92       * 
93       * <pre class="code">
94       * User user = ConstructorUtil.newInstance(User.class)
95       * </pre>
96       * 
97       * 调用无参的构造函数,返回User对象,你还可以:
98       * 
99       * <pre class="code">
100      * 
101      * User user1 = ConstructorUtil.newInstance(User.class, 100L);
102      * </pre>
103      * 
104      * 返回 id 是100的user对象构造函数
105      * </blockquote>
106      * 
107      * @param <T>
108      *            the generic type
109      * @param klass
110      *            类
111      * @param parameterValues
112      *            构造函数的参数值, 比如100L
113      * @return 如果 <code>klass</code> 是null,抛出 {@link NullPointerException}<br>
114      *         有任何异常(比如 NoSuchMethodException 找不到相关参数的构造函数),将抛出 {@link ReflectException}
115      * @see com.feilong.core.lang.ClassUtil#toClass(Object...)
116      * @see java.lang.Class#getConstructor(Class...)
117      * @see java.lang.reflect.Constructor#newInstance(Object...)
118      * @see org.apache.commons.lang3.reflect.ConstructorUtils#invokeConstructor(Class, Object...)
119      * @see "org.springframework.beans.BeanUtils.instantiateClass(Constructor<T>, Object...)"
120      */
121     public static <T> T newInstance(Class<T> klass,Object...parameterValues){
122         Validate.notNull(klass, "klass can't be null!");
123 
124         Class<?>[] parameterTypes = ClassUtil.toClass(parameterValues);
125         return newInstance(klass, parameterValues, parameterTypes);
126     }
127 
128     //---------------------------------------------------------------
129 
130     /**
131      * 返回指定类型 <code>klass</code>,指定参数 <code>parameterValues</code> 和指定参数类型 <code>parameterTypes</code>的构造函数示例.
132      * 
133      * <h3>说明:</h3>
134      * <blockquote>
135      * <ol>
136      * <li>定位并调用一个构造函数。构造函数的签名必须与赋值兼容的参数类型相匹配。</li>
137      * <li>和 {@link #newInstance(Class, Object...)}的区别, 在于 如果一个类有些重载的构造函数,并且参数类型相似, 此时使用这个方法可以精准定位到需要的构造函数</li>
138      * </ol>
139      * </blockquote>
140      * 
141      * <h3>示例:</h3>
142      * 
143      * <blockquote>
144      * 
145      * <pre class="code">
146      * User user = ConstructorUtil.newInstance(User.class,null,null)
147      * </pre>
148      * 
149      * 调用无参的构造函数,返回User对象,你还可以:
150      * 
151      * <pre class="code">
152      * 
153      * User user1 = ConstructorUtil.newInstance(User.class, toArray(100L), toArray(Long.class));
154      * </pre>
155      * 
156      * 返回 id 是100的user对象构造函数
157      * </blockquote>
158      *
159      * @param <T>
160      *            the generic type
161      * @param klass
162      *            the class to be constructed, not {@code null}
163      * @param parameterValues
164      *            the array of arguments, {@code null} treated as empty
165      * @param parameterTypes
166      *            the array of parameter types, {@code null} treated as empty
167      * @return 如果 <code>klass</code> 是null,抛出 {@link NullPointerException}<br>
168      *         有任何异常(比如 NoSuchMethodException 找不到相关参数的构造函数),将抛出 {@link ReflectException}
169      * @see org.apache.commons.lang3.reflect.ConstructorUtils#invokeConstructor(Class, Object[], Class[])
170      * @see "org.springframework.beans.BeanUtils.instantiateClass(Constructor<T>, Object...)"
171      */
172     public static <T> T newInstance(Class<T> klass,Object[] parameterValues,Class<?>[] parameterTypes){
173         Validate.notNull(klass, "klass can't be null!");
174 
175         //---------------------------------------------------------------
176         try{
177             return ConstructorUtils.invokeConstructor(klass, parameterValues, parameterTypes);
178         }catch (Exception e){
179             String pattern = "invokeConstructor exception,class:[{}].args:[{}],parameterTypes:[{}]";
180             String message = Slf4jUtil.format(pattern, klass, parameterValues, parameterTypes);
181             throw new ReflectException(message, e);
182         }
183     }
184 }
