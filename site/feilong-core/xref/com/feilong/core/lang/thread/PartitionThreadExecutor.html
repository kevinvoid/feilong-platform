16  package com.feilong.core.lang.thread;
17  
18  import java.util.List;
19  import java.util.Map;
20  
21  /**
22   * 分区线程执行器.
23   * 
24   * <h3>背景:</h3>
25   * 
26   * <blockquote>
27   * 
28   * <p>
29   * 经常有如此的需求, 下载客户8000张图片加工处理, 如果串行执行耗时很长,此时需要使用多线程,<br>
30   * 但是如果从头开始写会开发蛮久时间,此时把核心功能进行封装 ,你只需要传入 8000和url list, 告知每个线程处理多少张图片, 以及每个线程的代码处理方式即可,实现类会自动分区创建线程并执行
31   * </p>
32   * 
33   * <p>
34   * 契合 feilong 的思想, Reduce development, Release ideas (减少开发,释放思想)
35   * </p>
36   * </blockquote>
37   * 
38   * <h3>快速调用以及实现类:</h3>
39   * 
40   * <blockquote>
41   * <p>
42   * 目前已知快速调用有 {@link com.feilong.core.lang.ThreadUtil#execute(List, int, Map, PartitionRunnableBuilder) }, 或者调用
43   * {@link com.feilong.core.lang.thread.DefaultPartitionThreadExecutor}, 或者调用
44   * {@link "com.feilong.spring.scheduling.concurrent.AsyncTaskExecutorPartitionThreadExecutor"}
45   * </p>
46   * 
47   * </blockquote>
48   * 
49   * <h3>如果使用spring asyncTaskExecutor:</h3>
50   * <blockquote>
51   * 
52   * <pre>
53  {@code
54   <!-- 配置线程池 -->
55      <bean id="taskExecutor" class="org.springframework.scheduling.concurrent.ThreadPoolTaskExecutor">
56           <!-- 线程池维护线程的最少数量 -->
57          <property name="corePoolSize" value="10" />
58           <!-- 线程池维护线程所允许的空闲时间 -->
59          <property name="keepAliveSeconds" value="30000" />
60           <!-- 线程池维护线程的最大数量 -->
61          <property name="maxPoolSize" value="1000" />
62           <!-- 线程池所使用的缓冲队列 -->
63          <property name="queueCapacity" value="200" />
64      </bean>
65      
66      <!-- partitionThreadExecutor -->
67  
68      <bean id="partitionThreadExecutor" class="com.feilong.spring.scheduling.concurrent.AsyncTaskExecutorPartitionThreadExecutor">
69          <property name="asyncTaskExecutor" ref="taskExecutor"></property>
70      </bean>
71      
72      }
73   * </pre>
74   * 
75   * 然后 java 代码调用
76   * 
77   * <pre>
78   * 
79   * <code>@Autowired</code>
80   * private PartitionThreadExecutor partitionThreadExecutor;
81   * 
82   * <code>@Test</code>
83   * public void test(){
84   *     final List{@code <Integer>} list = toList(1, 2, 3, 4, 5);
85   * 
86   *     partitionThreadExecutor.execute(list, 2, new PartitionRunnableBuilder{@code <Integer>}(){
87   * 
88   *         public Runnable build(List{@code <Integer>} perBatchList,final PartitionThreadEntity partitionThreadEntity,Map{@code <String, ?>} paramsMap){
89   *             return new Runnable(){
90   * 
91   *                 public void run(){
92   *                     for (Integer integer : list){
93   *                         LOGGER.debug("{}-{}", partitionThreadEntity.getBatchNumber(), integer);
94   *                     }
95   *                 }
96   *             };
97   *         }
98   *     });
99   * }
100  * </pre>
101  * 
102  * </blockquote>
103  *
104  * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
105  * @since 1.11.0
106  * @since 2.0.0 move from package com.feilong.core.lang
107  */
108 public interface PartitionThreadExecutor{
109 
110     /**
111      * 给定一个待解析的 <code>list</code>,设定每个线程执行多少条 <code>eachSize</code> ,使用自定义的
112      * <code>partitionRunnableBuilder</code>,自动<span style="color:green">构造多条线程</span>并运行.
113      * 
114      * <p>
115      * 如果 <code>list</code> 是null,抛出 {@link NullPointerException}<br>
116      * 如果 <code>list</code> 是empty,抛出 {@link IllegalArgumentException}<br>
117      * 如果 {@code eachSize <=0} ,抛出 {@link IllegalArgumentException}<br>
118      * 如果 <code>partitionRunnableBuilder</code> 是null,抛出 {@link NullPointerException}<br>
119      * </p>
120      * 
121      * @param <T>
122      *            the generic type
123      * @param list
124      *            执行解析的list
125      * 
126      *            <p>
127      *            比如 100000个 User,不能为null或者empty
128      *            </p>
129      * @param eachSize
130      *            每个线程执行多少个对象
131      * 
132      *            <p>
133      *            比如 一个线程解析 1000个 User, 那么程序内部 会自动创建 100000/1000个 线程去解析;<br>
134      *            必须{@code >}0
135      *            </p>
136      * @param partitionRunnableBuilder
137      *            每个线程做的事情,不能为null
138      * @see com.feilong.core.lang.ThreadUtil#execute(List, int, Map, PartitionRunnableBuilder)
139      * @since 2.0.0 change name from <b>excute</b> to <b>execute</b>
140      */
141     <T> void execute(List<T> list,int eachSize,PartitionRunnableBuilder<T> partitionRunnableBuilder);
142 
143     //---------------------------------------------------------------
144 
145     /**
146      * 给定一个待解析的 <code>list</code>,设定每个线程执行多少条 <code>eachSize</code>,传入一些额外的参数 <code>paramsMap</code>,使用自定义的
147      * <code>partitionRunnableBuilder</code>,自动<span style="color:green">构造多条线程</span>并运行.
148      * 
149      * <p>
150      * 如果 <code>list</code> 是null,抛出 {@link NullPointerException}<br>
151      * 如果 <code>list</code> 是empty,抛出 {@link IllegalArgumentException}<br>
152      * 如果 {@code eachSize <=0} ,抛出 {@link IllegalArgumentException}<br>
153      * 如果 <code>partitionRunnableBuilder</code> 是null,抛出 {@link NullPointerException}<br>
154      * </p>
155      * 
156      * @param <T>
157      *            the generic type
158      * @param list
159      *            执行解析的list
160      * 
161      *            <p>
162      *            比如 100000个 User,不能为null或者empty
163      *            </p>
164      * @param eachSize
165      *            每个线程执行多少个对象
166      * 
167      *            <p>
168      *            比如 一个线程解析 1000个 User, 那么程序内部 会自动创建 100000/1000个 线程去解析;<br>
169      *            必须{@code >}0
170      *            </p>
171      * @param paramsMap
172      *            自定义的相关参数
173      * 
174      *            <p>
175      *            该参数目的是你可以在自定义的 <code>partitionRunnableBuilder</code>中使用;<br>
176      *            如果你传入的<code>partitionRunnableBuilder</code>中不需要额外的自定义参数,那么此处可以传入null
177      *            </p>
178      * @param partitionRunnableBuilder
179      *            每个线程做的事情,不能为null
180      * @see com.feilong.core.lang.ThreadUtil#execute(List, int, Map, PartitionRunnableBuilder)
181      * @since 2.0.0 change name from <b>excute</b> to <b>execute</b>
182      */
183     <T> void execute(List<T> list,int eachSize,Map<String, ?> paramsMap,PartitionRunnableBuilder<T> partitionRunnableBuilder);
184 
185 }
