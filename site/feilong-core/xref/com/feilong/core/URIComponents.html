16  package com.feilong.core;
17  
18  /**
19   * uri中使用的字符常量.
20   * 
21   * @author <a href="http://feitianbenyue.iteye.com/">feilong</a>
22   * @see "org.springframework.web.util.HierarchicalUriComponents.Type"
23   * @see "org.springframework.web.util.UriComponentsBuilder"
24   * @see "org.springframework.web.util.UriComponents"
25   * @since 1.0.9
26   */
27  public final class URIComponents{
28  
29      /** 查询片段 <code>{@value}</code>. */
30      public static final String FRAGMENT     = "#";
31  
32      /** <code>{@value}</code> The question mark is used as a separator and is not part of the query string. */
33      public static final String QUESTIONMARK = "?";
34  
35      /** The Constant ampersand<code>{@value}</code>. */
36      public static final String AMPERSAND    = "&";
37  
38      //---------------------------------------------------------------
39  
40      /**
41       * http协议<code>{@value}</code>.
42       * 
43       * <p>
44       * 超文本传输协议(HTTP,HyperText Transfer Protocol)是互联网上应用最为广泛的一种网络协议。
45       * </p>
46       */
47      public static final String SCHEME_HTTP  = "http";
48  
49      /**
50       * https协议<code>{@value}</code>.
51       * 
52       * <p>
53       * 全称：Hyper Text Transfer Protocol over Secure Socket Layer; <br>
54       * 是以安全为目标的HTTP通道,简单讲是HTTP的安全版。<br>
55       * 即HTTP下加入SSL层,HTTPS的安全基础是SSL,因此加密的详细内容就需要SSL。
56       * </p>
57       * 
58       * <h3>HTTPS和HTTP的区别主要为以下四点:</h3>
59       * 
60       * <blockquote>
61       * <ol>
62       * <li>https协议需要到ca申请证书,一般免费证书很少,需要交费。</li>
63       * <li>http是超文本传输协议,信息是明文传输,https 则是具有安全性的ssl加密传输协议。</li>
64       * <li>http和https使用的是完全不同的连接方式,用的端口也不一样,前者是80,后者是443。</li>
65       * <li>http的连接很简单,是无状态的；HTTPS协议是由SSL+HTTP协议构建的可进行加密传输、身份认证的网络协议,比http协议安全。</li>
66       * </ol>
67       * </blockquote>
68       */
69      public static final String SCHEME_HTTPS = "https";
70  
71      //---------------------------------------------------------------
72  
73      /** Don't let anyone instantiate this class. */
74      private URIComponents(){
75          //AssertionError不是必须的. 但它可以避免不小心在类的内部调用构造器. 保证该类在任何情况下都不会被实例化.
76          //see 《Effective Java》 2nd
77          throw new AssertionError("No " + getClass().getName() + " instances for you!");
78      }
79  }
