1   package com;

3   import java.util.Calendar;
4   import java.util.Date;
5   import java.util.GregorianCalendar;
6   
7   import org.quartz.Job;
8   import org.quartz.JobExecutionContext;
9   import org.quartz.JobExecutionException;
10  
11  public class SimpleJobDemo1 implements Job{
12  
13      private final Calendar calendar = new GregorianCalendar();
14  
15      @Override
16      public void execute(JobExecutionContext context) throws JobExecutionException{
17          calendar.setTime(new Date());
18          int year = calendar.get(Calendar.YEAR);
19          int month = calendar.get(Calendar.MONTH);
20          int day = calendar.get(Calendar.DAY_OF_MONTH);
21          int hh = calendar.get(Calendar.HOUR_OF_DAY);
22          int mi = calendar.get(Calendar.MINUTE);
23          int ss = calendar.get(Calendar.SECOND);
24          String s = year + "-" + month + "-" + day + " " + hh + ":" + mi + ":" + ss;
25          System.out.println(
26                          s + ",triggerName:" + context.getTrigger().getKey().getName() + ",triggerGroup:"
27                                          + context.getTrigger().getKey().getName() + ",jobName:" + context.getJobDetail().getKey().getName()
28                                          + ",jobGroup:" + context.getJobDetail().getKey().getGroup() + ",jobMap:"
29                                          + context.getTrigger().getJobDataMap().getWrappedMap());
30      }
31  }
