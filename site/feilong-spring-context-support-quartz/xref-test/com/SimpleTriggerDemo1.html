18  public class SimpleTriggerDemo1{
19  
20      public static void main(String[] args) throws SchedulerException{
21          new SimpleTriggerDemo1().run();
22      }
23  
24      /**
25       * quartz的类写的很优雅，类的方法能进行连写，和jQuery很像,好处很明显，优雅的同时也减少了代码，提高了创建类对象的使用
26       * 
27       * quartz包含job和trigger两个概念，job就是你要做的事情，trigger当然是触发器，他们都是有名称和组的，并且还有参数，
28       * 这个参数呢其实就是一个quartz自己搞的一个继承了java.util.Map的类org.quartz.JobDataMap。
29       * 基本上org.quartz.*包能完成大部分的定时器的任务，因为初学这个所以今天也就只看了这个包的内容，感觉还是非常容易伤手的。
30       * 总结：org.quartz.*包下基本上搞懂job、trigger、scheduler这3个相关的东西就行了，因为定时相关的就是围绕这3个东西展开的。
31       * 
32       * @throws SchedulerException
33       */
34      public void run() throws SchedulerException{
35          SchedulerFactory facroty = new StdSchedulerFactory();
36          Scheduler scheduler = facroty.getScheduler();
37  
38          JobDetail jobDetail = buildJobDetail();
39          scheduler.scheduleJob(jobDetail, buildTrigger1());
40          scheduler.scheduleJob(buildTrigger2(jobDetail));//同一个job添加多个trigger  
41          scheduler.start();
42      }
43  
44      private JobDetail buildJobDetail(){
45          return JobBuilder.newJob(SimpleJobDemo1.class).withIdentity(new JobKey("job名称", "job组名称")).build();
46      }
47  
48      private Trigger buildTrigger1(){
49          JobDataMap newJobDataMap = new JobDataMap();
50          newJobDataMap.put("key1", "value1");
51          newJobDataMap.put("key2", "value2");
52  
53          return TriggerBuilder.newTrigger()
54                          // 添加身份标识，可以添加或不添加  
55                          .withIdentity(new TriggerKey("trigger名称", "trigger组名称"))
56                          .withSchedule(SimpleScheduleBuilder.simpleSchedule().withIntervalInSeconds(1).repeatForever())
57                          //                  .withSchedule(CronScheduleBuilder.cronSchedule(""))  
58                          .usingJobData(newJobDataMap).startAt(
59                                          // future将来的意思，futureDate就是将来的某个时间执行  
60                                          DateBuilder.futureDate(2, IntervalUnit.SECOND))
61                          .build();
62      }
63  
64      private Trigger buildTrigger2(JobDetail jobDetail){
65          return TriggerBuilder.newTrigger()
66                          // 添加身份标识，可以添加或不添加  
67                          .withIdentity(new TriggerKey("trigger名称2", "trigger组名称2"))
68                          .withSchedule(SimpleScheduleBuilder.simpleSchedule().withIntervalInSeconds(2).repeatForever()).startAt(
69                                          // future将来的意思，futureDate就是将来的某个时间执行  
70                                          DateBuilder.futureDate(5, IntervalUnit.SECOND))
71                          .forJob(jobDetail).build();
72      }
73  }
